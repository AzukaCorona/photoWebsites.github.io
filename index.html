<h3>Artist Statement</h3>

<p>In his art, Azuka Corona captures light over time to turn pictures into a medium of art more similar to painting. With the use of longer exposure times, it is possible to essentially print certain images onto the screen in any place within frame. Azuka utilizes this ability to make possible things normally seen as impossible such as the duplication of a person or the appearance of a person in a place they never were. The opposite of this type of photography completely stills the frame only showing exactly what exists in the moment the photo was taken. This can be used to capture light as it was in a specific moment. Azuka uses this more standard type of photography when capturing photos of people outside around him to get a brief snapshot into people’s lives. The exploration of different ways to use photography, in both aspects of painting a photo, and taking an exact snapshot of the world, as well as the many other applications is what inspires Azuka to take his photos.

</p>

<h3>Artist Bio</h3>

<p>Azuka is extremely interested in the technical aspects of photography and the possibilities for the use of movement, combined with possible camera modifications and movements to create either an artistic vision or a document of the world. Having been extremely invested in the scientific worlds of chemistry and biology, documenting the world through both the lens of people and objects is an integral part of Azuka’s photography. On the other side of photography, having taken many visual arts classes throughout his life Azuka also likes to express his creativity through his experimental use of camera features to create unique photos which can more readily express his imaginative thoughts in the way that paintings, drawings, and sculptures can. The dichotomy between these two artistic inspirations results in the diverse set of photos shown in his portfolio.

</p>

<script>

const items = document.querySelectorAll('.gallery-item');

const images = Array.from(document.querySelectorAll('.gallery-grid img'));

const lightbox = document.getElementById('lightbox');

const imgContainer = document.getElementById('img-container');

const lightboxClose = document.getElementById('lightbox-close');

const lightboxCounter = document.getElementById('lightbox-counter');

const btnPrev = document.getElementById('lightbox-prev');

const btnNext = document.getElementById('lightbox-next');

let currentIndex = 0;

let isAnimating = false; // Flag to prevent rapid clicks from breaking the transition pipeline


// Helper to setup number strings gracefully

function updateCounter(index) {

const displayNum = String(index + 1).padStart(2, '0');

const totalNum = String(images.length).padStart(2, '0');

lightboxCounter.textContent = `${displayNum} / ${totalNum}`;

}


// Creates a clean image element styled for our lightbox

function createLightboxImage(imgElement) {

const el = document.createElement('img');

el.className = 'lightbox-img';

el.src = imgElement.src;

el.alt = imgElement.alt;

return el;

}


// Handles initial lightbox opening

items.forEach((item, index) => {

item.addEventListener('click', () => {

currentIndex = index;

imgContainer.innerHTML = ''; // Wipe out any lingering elements

const initialImg = createLightboxImage(images[currentIndex]);

imgContainer.appendChild(initialImg);

updateCounter(currentIndex);

lightbox.classList.add('active');

});

});


function switchImage(direction) {

if (isAnimating) return;

isAnimating = true;


let targetIndex;

if (direction === 'next') {

targetIndex = (currentIndex + 1) % images.length;

} else {

targetIndex = (currentIndex - 1 + images.length) % images.length;

}


const currentImgElement = imgContainer.querySelector('.lightbox-img');

const nextImgElement = createLightboxImage(images[targetIndex]);


// Phase Stage 1: Setup entrance configurations based on direction paths

if (direction === 'next') {

nextImgElement.classList.add('slide-in-right');

} else {

nextImgElement.classList.add('slide-in-left');

}


imgContainer.appendChild(nextImgElement);


// Phase Stage 2: Force reflow layout refresh so the browser registers the entry states properly

nextImgElement.getBoundingClientRect();


// Phase Stage 3: Trigger transitions simultaneously

if (direction === 'next') {

currentImgElement.classList.add('slide-out-left');

} else {

currentImgElement.classList.add('slide-out-right');

}

nextImgElement.classList.remove('slide-in-right', 'slide-in-left');


// Update indices data states mid flight

currentIndex = targetIndex;

updateCounter(currentIndex);


// Phase Stage 4: Clean old image references out when transition ends

setTimeout(() => {

if (currentImgElement.parentNode === imgContainer) {

imgContainer.removeChild(currentImgElement);

}

isAnimating = false;

}, 550); // Matches the 0.55s CSS transition speed perfectly

}


function closeLightbox() {

lightbox.classList.remove('active');

setTimeout(() => {

imgContainer.innerHTML = ''; // Safely drop DOM elements out when closed

}, 500);

}


lightboxClose.addEventListener('click', closeLightbox);

btnNext.addEventListener('click', (e) => { e.stopPropagation(); switchImage('next'); });

btnPrev.addEventListener('click', (e) => { e.stopPropagation(); switchImage('prev'); });


lightbox.addEventListener('click', (e) => {

if (e.target === lightbox || e.target.classList.contains('lightbox-content-wrapper')) {

closeLightbox();

}

});

document.addEventListener('keydown', (e) => {

if (!lightbox.classList.contains('active')) return;

if (e.key === 'Escape') closeLightbox();

if (e.key === 'ArrowRight') switchImage('next');

if (e.key === 'ArrowLeft') switchImage('prev');

});

</script>
